```typescript
import { useEffect, useState } from 'react'

type VersionInfo = {
  version: string
  commit: string
  buildDate: string
}

export function useVersion(): VersionInfo | null {
  const [info, setInfo] = useState<VersionInfo | null>(null)
  useEffect(() => {
    let mounted = true
    fetch('/api/version')
      .then((r) => (r.ok ? r.json() : null))
      .then((d: VersionInfo | null) => {
        if (mounted && d) setInfo(d)
      })
      .catch(() => {})
    return () => {
      mounted = false
    }
  }, [])
  return info
}
```

Create frontend/src/features/version/VersionBadge.tsx:

```tsx
import { useVersion } from './useVersion'

export function VersionBadge() {
  const info = useVersion()
  if (!info) return null
  const short = info.commit !== 'none' ? info.commit.slice(0, 7) : null
  const label = short ? `v${info.version} (${short})` : `v${info.version}`
  return (
    <span
      className="whitespace-nowrap text-[0.66rem] font-medium leading-none text-[#444]"
      aria-label={`Application version: ${label}`}
    >
      {label}
    </span>
  )
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GET /api/version → React component | Version data rendered into DOM; must not be trusted as authoritative for security decisions |
| fetch() in browser → backend | Standard CORS applies; hostHeader middleware (Plan 01) protects the endpoint |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Information Disclosure | VersionBadge displaying commit hash | accept | Commit hash is build provenance, not a secret. Local-only tool. Acceptable per ASVS L1 for localhost use. |
| T-06-02 | Spoofing | Malicious `/api/version` response rendering XSS | accept | React renders the version string as text (template literal into `{label}` JSX text node), not dangerouslySetInnerHTML. XSS is not possible via JSX text interpolation. |
| T-06-03 | Denial of Service | useVersion fetch() failing and leaving component stuck | accept | On error, useVersion returns null and VersionBadge renders nothing. No error state shown. No retry loop. No impact on app functionality. |
</threat_model>
