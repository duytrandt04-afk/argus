Create .github/workflows/ci.yml with this exact content:

```yaml
name: CI

on:
  push:
  pull_request:

jobs:
  backend:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version-file: backend/go.mod
          cache-dependency-path: backend/go.sum

      - name: Setup pnpm via corepack
        run: |
          corepack enable
          corepack prepare pnpm@10.23.0 --activate

      - name: Build frontend (required for go:embed)
        working-directory: frontend
        run: |
          pnpm install --frozen-lockfile
          pnpm run build

      - name: Build backend (verifies embed compiles)
        working-directory: backend
        run: go build ./...

      - name: Test
        working-directory: backend
        run: go test ./...

      - name: Vet
        working-directory: backend
        run: go vet ./...

      - name: govulncheck
        working-directory: backend
        run: |
          go install golang.org/x/vuln/cmd/govulncheck@latest
          govulncheck ./...
        continue-on-error: true

      - name: golangci-lint
        uses: golangci/golangci-lint-action@v6
        with:
          working-directory: backend
          version: v1.64

  frontend:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Setup pnpm via corepack
        run: corepack enable && corepack prepare pnpm@10.23.0 --activate

      - name: Install dependencies
        working-directory: frontend
        run: pnpm install --frozen-lockfile

      - name: Typecheck
        working-directory: frontend
        run: pnpm run typecheck

      - name: Test
        working-directory: frontend
        run: pnpm run test -- --run

      - name: Build
        working-directory: frontend
        run: pnpm run build
```

Create .goreleaser.yaml at the repository root with this exact content:

```yaml
# GoReleaser OSS v2 configuration
# Docs: https://goreleaser.com
version: 2

before:
  hooks:
    # Build frontend first — the Go binary embeds frontend/dist/* via //go:embed in
    # backend/internal/ui/ui.go. GoReleaser's own dist/ output is a different directory.
    - bash -c "cd frontend && pnpm install --frozen-lockfile && pnpm run build"

builds:
  - id: hooker
    main: ./cmd/server
    dir: backend
    binary: hooker
    env:
      - CGO_ENABLED=0
    goos:
      - linux
      - darwin
    goarch:
      - amd64
      - arm64
    flags:
      - -trimpath
    ldflags:
      - -s -w
      # Package path matches module: hooker/internal/version (NOT main.version)
      - -X hooker/internal/version.Version={{.Version}}
      - -X hooker/internal/version.Commit={{.Commit}}
      - -X hooker/internal/version.BuildDate={{.Date}}

archives:
  - id: default
    name_template: "hooker_{{ .Version }}_{{ .Os }}_{{ .Arch }}"
    format: tar.gz

checksum:
  name_template: "checksums.txt"

changelog:
  use: github
  sort: asc
  filters:
    exclude:
      - "^docs:"
      - "^test:"
      - "^ci:"
      - "Merge pull request"
```

Create .github/workflows/release.yml with this exact content:

```yaml
name: Release

on:
  push:
    tags:
      - 'v*'

permissions:
  contents: write

jobs:
  release:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - uses: actions/setup-go@v5
        with:
          go-version-file: backend/go.mod
          cache-dependency-path: backend/go.sum

      - uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Setup pnpm via corepack
        run: corepack enable && corepack prepare pnpm@10.23.0 --activate

      - name: Build frontend (per D-11 — explicit inline step required in every workflow)
        working-directory: frontend
        run: |
          pnpm install --frozen-lockfile
          pnpm run build

      - name: Release via GoReleaser
        uses: goreleaser/goreleaser-action@v6
        with:
          distribution: goreleaser
          version: "~> v2"
          args: release --clean
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

Important notes about release.yml:
- `fetch-depth: 0` is required by GoReleaser to generate the changelog from git history
- `permissions: contents: write` is required to create GitHub Releases
- `args: release --clean` uses --clean (NOT --rm-dist which is deprecated in v2)
- The explicit "Build frontend" step satisfies D-11 (frontend build must be an inline step
  in every workflow). GoReleaser's before.hooks also runs pnpm build — the explicit step
  ensures the dist/ is present in the runner environment before GoReleaser begins, and makes
  the dependency visible at a glance in the workflow file.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner → repository | CI runs untrusted code on push/PR; supply chain attack surface |
| GITHUB_TOKEN | Used to create releases; scoped to contents: write only |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | CI running `go install govulncheck@latest` | accept | Fetches latest version at CI time; pinned `@latest` means it could change. Acceptable for advisory-only vuln scan. Risk is false confidence, not code compromise. |
| T-03-02 | Elevation of Privilege | `GITHUB_TOKEN` in release.yml | mitigate | `permissions: contents: write` scopes the token to the minimum needed for release creation. Default token cannot write to other repos. |
| T-03-03 | Tampering | GoReleaser before.hooks running arbitrary commands | accept | before.hooks runs `pnpm install --frozen-lockfile` — lockfile prevents dependency tampering. `--frozen-lockfile` is the mitigation. |
| T-03-04 | Spoofing | Release binaries not signed | accept | Phase 1 scope: checksums.txt provides integrity verification without code signing. Signed artifacts are v2 backlog. Document in release runbook. |
</threat_model>
