```bash
HOOKER_HOOK_CMD="curl -s --max-time 2 -X POST http://127.0.0.1:8765/api/hook -H 'Content-Type: application/json' -d @- || true"

patch_claudecode_hooks() {
  local settings="$HOME/.claude/settings.json"
  if [ ! -f "$settings" ]; then
    printf '[skip] Claude Code settings.json not found at %s\n' "$settings"
    return 0
  fi
  if grep -q "8765/api/hook" "$settings"; then
    printf '[ok]   Claude Code hooks already configured — skipping\n'
    return 0
  fi
  cp "$settings" "${settings}.bak.pre-hooker"
  python3 - "$settings" "$HOOKER_HOOK_CMD" <<'PYEOF'
import sys, json
path, cmd = sys.argv[1], sys.argv[2]
with open(path) as f:
    d = json.load(f)
hooks = d.setdefault('hooks', {})
entry = {'hooks': [{'type': 'command', 'command': cmd}]}
for event in ['PreToolUse', 'PostToolUse', 'SessionStart', 'SessionEnd', 'Stop']:
    hooks.setdefault(event, []).append(entry)
with open(path, 'w') as f:
    json.dump(d, f, indent=2)
PYEOF
  printf '[ok]   Claude Code hooks configured (backup: %s.bak.pre-hooker)\n' "$settings"
}

patch_codex_hooks() {
  local settings="$HOME/.codex/hooks.json"
  if [ ! -f "$settings" ]; then

  if grep -q "8765/api/hook" "$settings"; then

}
```

The new doctor() function:

```bash
doctor() {
  local doctor_errors=0
  local PASS='[ok]  '
  local FAIL='[FAIL]'
  local WARN='[warn]'

  check_required() {
    local label="$1" result="$2" hint="$3"
    if [ "$result" = "pass" ]; then
      printf '%s %s\n' "$PASS" "$label"
    else
      printf '%s %s\n      fix: %s\n' "$FAIL" "$label" "$hint"
      doctor_errors=$((doctor_errors + 1))
    fi
  }

  check_optional() {
    local label="$1" result="$2" hint="$3"
    if [ "$result" = "pass" ]; then
      printf '%s %s\n' "$PASS" "$label"
    else
      printf '%s %s\n      note: %s\n' "$WARN" "$label" "$hint"
    fi
  }

  printf 'hooker doctor — required checks\n'
  printf '────────────────────────────────\n'

  # Go version >= 1.25.0
  if command -v go >/dev/null 2>&1; then
    go_ver=$(go version | awk '{print $3}' | sed 's/go//')
    go_major=$(printf '%s' "$go_ver" | cut -d. -f1)
    go_minor=$(printf '%s' "$go_ver" | cut -d. -f2)
    if [ "$go_major" -gt 1 ] || { [ "$go_major" -eq 1 ] && [ "$go_minor" -ge 25 ]; }; then
      check_required "go >= 1.25.0 (found $go_ver)" "pass" ""
    else
      check_required "go >= 1.25.0 (found $go_ver)" "fail" "Install Go 1.25.0 or newer from https://go.dev/dl/"
    fi
  else
    check_required "go installed" "fail" "Install Go 1.25.0 or newer from https://go.dev/dl/"
  fi

  # Node version >= 18
  if command -v node >/dev/null 2>&1; then
    node_ver=$(node --version | sed 's/v//')
    node_major=$(printf '%s' "$node_ver" | cut -d. -f1)
    if [ "$node_major" -ge 18 ]; then
      check_required "node >= 18 (found v$node_ver)" "pass" ""
    else
      check_required "node >= 18 (found v$node_ver)" "fail" "Install Node.js 18 or newer from https://nodejs.org/"
    fi
  else
    check_required "node installed" "fail" "Install Node.js 18 or newer from https://nodejs.org/"
  fi

  # pnpm present
  if command -v pnpm >/dev/null 2>&1; then
    check_required "pnpm installed ($(pnpm --version))" "pass" ""
  else
    check_required "pnpm installed" "fail" "Run: corepack enable && corepack prepare pnpm@10.23.0 --activate"
  fi

  # DB path writable
  local db_path="${DB_PATH:-$BACKEND/hooker.db}"
  local db_dir
  db_dir="$(dirname "$db_path")"
  if [ -w "$db_dir" ]; then
    check_required "DB path writable ($db_path)" "pass" ""
  else
    check_required "DB path writable ($db_path)" "fail" "Directory $db_dir is not writable. Check permissions or set DB_PATH to a writable path."
  fi

  # Port 8765 available or hooker running
  if command -v curl >/dev/null 2>&1 && curl -fsS --max-time 1 http://127.0.0.1:8765/healthz >/dev/null 2>&1; then
    check_required "port 8765 (hooker running)" "pass" ""
  elif ! command -v lsof >/dev/null 2>&1 || ! lsof -i :8765 >/dev/null 2>&1; then
    check_required "port 8765 available" "pass" ""
  else
    check_required "port 8765 available" "fail" "Port 8765 is in use by another process. Run: lsof -i :8765"
  fi

  printf '\nhooker doctor — optional checks\n'
  printf '────────────────────────────────\n'

  # Claude Code hook config present
  local claude_cfg="$HOME/.claude/settings.json"
  if [ -f "$claude_cfg" ] && grep -q "8765/api/hook" "$claude_cfg"; then
    check_optional "Claude Code hooks configured" "pass" ""
  else
    check_optional "Claude Code hooks configured" "warn" "Run: ./scripts/hooker setup"
  fi

  # Codex hook config present
  local codex_cfg="$HOME/.codex/hooks.json"
  if [ -f "$codex_cfg" ] && grep -q "8765/api/hook" "$codex_cfg"; then
    check_optional "Codex hooks configured" "pass" ""
  else
    check_optional "Codex hooks configured" "warn" "Run: ./scripts/hooker setup"
  fi

  # Non-loopback ADDR warning (DIAG-06)
  local addr="${ADDR:-127.0.0.1:8765}"
  local addr_host
  addr_host="$(printf '%s' "$addr" | cut -d: -f1)"
  if [ "$addr_host" = "127.0.0.1" ] || [ "$addr_host" = "localhost" ] || [ "$addr_host" = "::1" ]; then
    check_optional "ADDR is loopback ($addr)" "pass" ""
  else
    check_optional "ADDR is loopback ($addr)" "warn" "Non-loopback bind exposes hooker on the network. Hooker stores prompts, file paths, and diffs. Keep ADDR on loopback for personal use."
  fi

  printf '\n'
  if [ "$doctor_errors" -gt 0 ]; then
    printf '%d required check(s) failed. Fix the issues above before using hooker.\n' "$doctor_errors"
    return 1
  else
    printf 'all required checks passed.\n'
  fi
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| scripts/hooker → ~/.claude/settings.json | Script writes to user's Claude Code config — wrong write corrupts agent hooks |
| scripts/hooker → ~/.codex/hooks.json | Same as above for Codex config |
| python3 heredoc | JSON parsing in shell; malformed JSON in settings file could cause python3 to fail |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Writing to ~/.claude/settings.json without backup | mitigate | `cp "$settings" "${settings}.bak.pre-hooker"` runs before any write. If python3 fails, the backup exists for manual recovery. |
| T-04-02 | Tampering | Idempotency failure adding duplicate hooks | mitigate | `grep -q "8765/api/hook"` check runs before any write. Existing entries are never duplicated. |
| T-04-03 | Information Disclosure | doctor() reading ADDR env var | accept | ADDR is a local env var set by the developer; no external disclosure. |
| T-04-04 | Denial of Service | setup() failing mid-patch (e.g. python3 absent) | accept | `set -euo pipefail` causes immediate exit on error. Backup was already created before python3 runs. User can restore from .bak file. |
</threat_model>
